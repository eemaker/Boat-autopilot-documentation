107  using std::atan;
108  real d = x - y, xy = x * y;
109  return d != 0 ?

113  static real Dsin(real x, real y) {

118  static real Dsinh(real x, real y) {

122  }
123  static real Dcosh(real x, real y) {
124  using std::sinh;
125  real d = (x - y) / 2;
126  return sinh((x + y) / 2) * (d != 0 ? sinh(d) / d : 1);
127  }
128  static real Dasinh(real x, real y) {
129  real d = x - y,
130  hx = Math::hypot(real(1), x), hy = Math::hypot(real(1), y);
131  return d != 0 ? Math::asinh(x*y > 0 ? d * (x + y) / (x*hy + y*hx) :
132  x*hy - y*hx) / d :
133  1 / hx;
134  }

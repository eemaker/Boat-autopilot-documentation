245  template<typename T> static T hypot(T x, T y) {
246 #if GEOGRAPHICLIB_CXX11_MATH
247  using std::hypot; return hypot(x, y);
248 #else
249  using std::abs; using std::sqrt;
250  x = abs(x); y = abs(y);
251  if (x < y) std::swap(x, y); // Now x >= y >= 0
252  y /= (x ? x : 1);

255  // C. Moler and D. Morrision (1983) https://doi.org/10.1147/rd.276.0577

267  template<typename T> static T expm1(T x) {
268 #if GEOGRAPHICLIB_CXX11_MATH
269  using std::expm1; return expm1(x);
270 #else
271  using std::exp; using std::abs; using std::log;
272  GEOGRAPHICLIB_VOLATILE T
273  y = exp(x),
274  z = y - 1;
275  // The reasoning here is similar to that for log1p. The expression
276  // mathematically reduces to exp(x) - 1, and the factor z/log(y) = (y -
277  // 1)/log(y) is a slowly varying quantity near y = 1 and is accurately
278  // computed.
279  return abs(x) > 1 ? z : (z == 0 ? x : x * z / log(y));
280 #endif

290  template<typename T> static T log1p(T x) {
291 #if GEOGRAPHICLIB_CXX11_MATH
292  using std::log1p; return log1p(x);
293 #else
294  using std::log;
295  GEOGRAPHICLIB_VOLATILE T
296  y = 1 + x,
297  z = y - 1;
298  // Here's the explanation for this magic: y = 1 + z, exactly, and z
299  // approx x, thus log(y)/z (which is nearly constant near z = 0) returns
300  // a good approximation to the true log(1 + x)/x. The multiplication x *
301  // (log(y)/z) introduces little additional error.
302  return z == 0 ? x : x * log(y) / z;
303 #endif

494  template<typename T> static T AngDiff(T x, T y, T& e) {
495 #if GEOGRAPHICLIB_CXX11_MATH && GEOGRAPHICLIB_PRECISION != 4
496  using std::remainder;
497  T t, d = AngNormalize(sum(remainder(-x, T(360)),
498  remainder(y, T(360)), t));
499 #else
500  T t, d = AngNormalize(sum(AngNormalize(-x), AngNormalize(y), t));
501 #endif
502  // Here y - x = d + t (mod 360), exactly, where d is in (-180,180] and
503  // abs(t) <= eps (eps = 2^-45 for doubles). The only case where the
504  // addition of t takes the result outside the range (-180,180] is d = 180
505  // and t > 0. The case, d = -180 + eps, t = -eps, can't happen, since
506  // sum would have returned the exact result in such a case (i.e., given t
507  // = 0).
508  return sum(d == 180 && t > 0 ? -180 : d, t, e);
509  }

568  template<typename T> static void sincosd(T x, T& sinx, T& cosx) {
569  // In order to minimize round-off errors, this function exactly reduces
570  // the argument to the range [-45, 45] before converting it to radians.
571  using std::sin; using std::cos;
572  T r; int q;
573 #if GEOGRAPHICLIB_CXX11_MATH && GEOGRAPHICLIB_PRECISION <= 3 && \
574  !defined(__GNUC__)
575  // Disable for gcc because of bug in glibc version < 2.22, see
576  // https://sourceware.org/bugzilla/show_bug.cgi?id=17569
577  // Once this fix is widely deployed, should insert a runtime test for the
578  // glibc version number. For example
579  // #include <gnu/libc-version.h>
580  // std::string version(gnu_get_libc_version()); => "2.22"
581  using std::remquo;
582  r = remquo(x, T(90), &q);
583 #else
584  using std::fmod; using std::floor;
585  r = fmod(x, T(360));
586  q = int(floor(r / 90 + T(0.5)));
587  r -= 90 * q;
588 #endif
589  // now abs(r) <= 45
590  r *= degree();
591  // Possibly could call the gnu extension sincos
592  T s = sin(r), c = cos(r);
593 #if defined(_MSC_VER) && _MSC_VER < 1900
594  // Before version 14 (2015), Visual Studio had problems dealing
595  // with -0.0. Specifically
596  // VC 10,11,12 and 32-bit compile: fmod(-0.0, 360.0) -> +0.0
597  // VC 12 and 64-bit compile: sin(-0.0) -> +0.0
598  // AngNormalize has a similar fix.
599  // python 2.7 on Windows 32-bit machines has the same problem.
600  if (x == 0) s = x;
601 #endif
602  switch (unsigned(q) & 3U) {
603  case 0U: sinx = s; cosx = c; break;
604  case 1U: sinx = c; cosx = -s; break;
605  case 2U: sinx = -s; cosx = -c; break;
606  default: sinx = -c; cosx = s; break; // case 3U
607  }
608  // Set sign of 0 results. -0 only produced for sin(-0)
609  if (x != 0) { sinx += T(0); cosx += T(0); }
610  }

701  template<typename T> static T atan2d(T y, T x) {
702  // In order to minimize round-off errors, this function rearranges the
703  // arguments so that result of atan2 is in the range [-pi/4, pi/4] before
704  // converting it to degrees and mapping the result to the correct
705  // quadrant.
706  using std::atan2; using std::abs;
707  int q = 0;
708  if (abs(y) > abs(x)) { std::swap(x, y); q = 2; }
709  if (x < 0) { x = -x; ++q; }
710  // here x >= 0 and x >= abs(y), so angle is in [-pi/4, pi/4]
711  T ang = atan2(y, x) / degree();
712  switch (q) {
713  // Note that atan2d(-0.0, 1.0) will return -0. However, we expect that
714  // atan2d will not be called with y = -0. If need be, include
715  //
716  // case 0: ang = 0 + ang; break;
717  //
718  // and handle mpfr as in AngRound.
719  case 1: ang = (y >= 0 ? 180 : -180) - ang; break;
720  case 2: ang = 90 - ang; break;
721  case 3: ang = -90 + ang; break;
722  }
723  return ang;
724  }
